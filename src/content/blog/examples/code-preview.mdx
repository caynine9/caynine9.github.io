### Go (Golang)

```go
package main

import "fmt"

func main() {
    messages := []string{"Hello", "Astro", "Golang"}
    for _, msg := range messages {
        fmt.Printf("%s from Go!\n", msg)
    }
}
```

### Kotlin

```kotlin
fun main() {
    val items = listOf("Apple", "Banana", "Kiwi")
    items.filter { it.startsWith("A") }
         .map { it.uppercase() }
         .forEach { println(it) }
}
```

### Java

```java
public class HelloWorld {
    public static void main(String[] args) {
        System.out.println("Hello, Java!");
        
        List<String> list = Arrays.asList("Spring", "Astro", "Maven");
        list.stream()
            .filter(s -> s.length() > 5)
            .forEach(System.out::println);
    }
}
```
